- .bash_profile

```
# TERMINAL customize
export CLICOLOR=1
export GREP_OPTIONS="--color=auto"

```

```
hardstatus alwayslastline
shelltitle 'bash'
hardstatus string "%{.bW}%-w%{.rW}%n*%t%{-}%+w %= %c ${USER}@%H"
```
